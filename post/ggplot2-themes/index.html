ggplot2: Themes

25 Jun 2016
ggplot2
ggplot2 / visualization

This is the last post in the series Elegant Data Visualization with ggplot2. In the previous post, we learnt to combine multiple plots. In this post, we will learn to

  • modify theme components
  • use different themes

Use theme() to modify the appearance of all non data components of the plot.

theme

modify components of a plot

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg))
# modify X axis title
p + theme(axis.title.x = element_text(
  color = "red", size = 10))

# modify X axis text
p + theme(axis.text.x = element_text(
  color = "red", size = 10))

# modify both axis text
p + theme(axis.text = element_text(
  color = "red", size = 10, face = "italic"))

# modify axis ticks
p + theme(
  axis.ticks = element_line(
    color = 'blue',
    size = 1.25,
    linetype = 2
  ),
  axis.ticks.length = unit(1, "cm")
)

modify axis line

p + theme(
  axis.line = element_line(
    color = 'red',
    size = 1.5, 
    linetype = 3
  )
)

modify legend

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg, color = factor(cyl), shape = factor(gear)))
p + theme(
  legend.background = element_rect(
    fill = 'gray', 
    linetype = 2, 
    color = "red"
  ),
  legend.margin = margin(4, 8, 4, 8),
  legend.spacing = unit(1.5, "cm"),
  legend.key = element_rect(fill = 'red'),
  legend.key.size = unit(0.4, "cm"),
  legend.text = element_text(color = 'green', face = 'italic'),
  legend.title = element_text(color = 'blue', face = 'bold'),
  legend.title.align = 0.1,
  legend.position = "top",
  legend.direction = "horizontal",
  legend.box = "horizontal",
  legend.box.background = element_rect(fill = "yellow"),
  legend.box.spacing = unit(1, "cm"),
  legend.box.margin = margin(4, 8, 4, 8),
  legend.box.just = "left"
)

modify panel

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg))
p + theme(
  panel.background = element_rect(fill = 'gray'),
  panel.border = element_rect(
    fill = NA, 
    color = 'red', 
    linetype = 1,
    size = 2
  ),
  panel.grid = element_line(
    color = 'blue',
    linetype = 2,
    size = 0.5
  )
)

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg))
p + theme(
  panel.grid.major.x = element_line(
    color = 'blue',
    linetype = 2,
    size = 0.5
  ),
  panel.grid.minor.y = element_line(
    color = 'red',
    linetype = 3, 
    size = 0.2
  )
)

modify plot area

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg)) + 
  ggtitle('Theme Modification')
p + theme(
  plot.background = element_rect(
    color = 'blue',
    fill = '#4682B4'
  ),
  plot.title = element_text(color = 'red'),
  plot.margin = unit(c(1, 1, 1, 1), "cm")
)

modify facets

p <- ggplot(mtcars) +
  geom_point(aes(disp, mpg)) + 
  facet_grid(~cyl)
p + theme(
  strip.background = element_rect(
    color = 'red',
    fill = '#4682B4'
  ),
  strip.text = element_text(
    face = 'italic',
    size = 8, 
    color = 'red'
  )
)
